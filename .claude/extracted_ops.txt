/**
 * lib/db.ts
 * Singleton SQL Server connection pool en query helpers.
 * Gebruikt de mssql driver met parameterized queries (geen SQL-injectie mogelijk).
 */

import sql from 'mssql';

// ---------------------------------------------------------------
// Connection pool configuratie vanuit .env
// ---------------------------------------------------------------

/**
 * Parst "SERVER\INSTANCE" of "SERVER" naar aparte config velden.
 */
function parseServer(raw: string): { server: string; instanceName?: string } {
  const [server, instanceName] = (raw || '').split('\\');
  return { server, instanceName: instanceName || undefined };
}

const { server, instanceName } = parseServer(process.env.DB_SERVER || '');

const dbConfig: sql.config = {
  server,
  database:  process.env.DB_DATABASE || 'Chatbotdb',
  user:      process.env.DB_USER     || '',
  password:  process.env.DB_PASSWORD || '',
  options: {
    instanceName,
    trustServerCertificate: true, // nodig voor dev zonder geldig SSL-certificaat
    enableArithAbort: true,
  },
  pool: {
    max: 10,
    min: 0,
    idleTimeoutMillis: 30_000,
  },
  connectionTimeout: 15_000,
  requestTimeout:    30_000,
};

// ---------------------------------------------------------------
// Singleton pool
// ---------------------------------------------------------------

let _pool: sql.ConnectionPool | null = null;

/**
 * Geeft de actieve connection pool terug.
 * Maakt bij eerste aanroep de verbinding aan.
 */
export async function getPool(): Promise<sql.ConnectionPool> {
  if (_pool?.connected) return _pool;

  try {
    _pool = await sql.connect(dbConfig);
    console.log('[DB] ✅ Verbonden met SQL Server');
    return _pool;
  } catch (err) {
    console.error('[DB] ❌ Verbinding mislukt:', err);
    throw err;
  }
}

export { sql };

// ---------------------------------------------------------------
// Voertuig queries
// ---------------------------------------------------------------

/** Haalt alle voertuigen op met locatieinfo. */
export async function getVehicles() {
  const pool = await getPool();
  const result = await pool.request().query(`
    SELECT v.id, v.brand, v.model, v.type, v.seats, v.engineCC,
           v.dailyPrice, v.availabilityStatus, v.locationId, v.lastUpdated,
           l.name AS locationName, l.city AS locationCity
    FROM Vehicles v
    LEFT JOIN RentalLocations l ON v.locationId = l.id
    ORDER BY v.brand, v.model
  `);
  return result.recordset;
}

/** Haalt één voertuig op via ID. */
export async function getVehicleById(id: number) {
  const pool = await getPool();
  const result = await pool.request()
    .input('id', sql.Int, id)
    .query('SELECT * FROM Vehicles WHERE id = @id');
  return result.recordset[0] ?? null;
}

// Locatie queries

/** Haalt alle actieve verhuurlocaties op. */
export async function getLocations() {
  const pool = await getPool();
  const result = await pool.request().query(`
    SELECT id, name, city, address, phone, email, isActive
    FROM   RentalLocations
    WHERE  isActive = 1
    ORDER  BY city
  `);
  return result.recordset;
}

/** Maakt een nieuwe conversatie aan en retourneert het conversationId. */
export async function createConversation(userId: string): Promise<string> {
  const pool = await getPool();
  const conversationId = `conv_${Date.now()}_${Math.random().toString(36).slice(2, 9)}`;
  await pool.request()
    .input('conversationId', sql.NVarChar(100), conversationId)
    .input('userId',         sql.NVarChar(100), userId)
    .input('timestamp',      sql.DateTime,      new Date())
    .query(`
      INSERT INTO Conversations (conversationId, userId, timestamp)
      VALUES (@conversationId, @userId, @timestamp)
    `);
  return conversationId;
}

/** Slaat één chatbericht op. */
export async function saveMessage(params: {
  conversationId: string;
  role: 'user' | 'assistant' | 'system';
  content: string;
  llmModelUsed?: string;
}) {
  const pool = await getPool();
  await pool.request()
    .input('conversationId', sql.NVarChar(100),  params.conversationId)
    .input('role',           sql.NVarChar(20),   params.role)
    .input('content',        sql.NVarChar(sql.MAX), params.content)
    .input('llmModelUsed',   sql.NVarChar(100),  params.llmModelUsed ?? null)
    .input('timestamp',      sql.DateTime,       new Date())
    .query(`
      INSERT INTO Messages (conversationId, role, content, llmModelUsed, timestamp)
      VALUES (@conversationId, @role, @content, @llmModelUsed, @timestamp)
    `);
}

/** Haalt de berichtgeschiedenis van een conversatie op. */
export async function getConversationMessages(conversationId: string) {
  const pool = await getPool();
  const result = await pool.request()
    .input('conversationId', sql.NVarChar(100), conversationId)
    .query(`
      SELECT messageId, role, content, timestamp, llmModelUsed
      FROM   Messages
      WHERE  conversationId = @conversationId
      ORDER  BY timestamp ASC
    `);
  return result.recordset;
}

/** Haalt alle conversaties op voor admin-overzicht. */
export async function getAllConversations(limit = 50) {
  const pool = await getPool();
  const result = await pool.request()
    .input('limit', sql.Int, limit)
    .query(`
      SELECT TOP (@limit)
             c.conversationId, c.userId, c.timestamp,
             COUNT(m.messageId) AS messageCount
      FROM   Conversations c
      LEFT   JOIN Messages m ON c.conversationId = m.conversationId
      GROUP  BY c.conversationId, c.userId, c.timestamp
      ORDER  BY c.timestamp DESC
    `);
  return result.recordset;
}

/**
 * Maakt een streaming aanroep naar LM Studio.
 * Retourneert een ReadableStream van SSE-events.
 * Elke data-chunk bevat { token: string } of [DONE] of { error: string }.
 */
export async function streamLLMResponse(
  messages: ChatMessage[],
  signal?: AbortSignal
): Promise<ReadableStream<Uint8Array>> {
  const llmUrl   = process.env.LLM_URL   || 'http://localhost:1234/v1/chat/completions';
  const llmModel = process.env.LLM_MODEL || 'local-model';
  const timeout  = Number(process.env.LLM_TIMEOUT ?? 30_000);
  const encoder  = new TextEncoder();

  return new ReadableStream({
    async start(controller) {
      const emit = (data: unknown) =>
        controller.enqueue(encoder.encode(`data: ${JSON.stringify(data)}\n\n`));

      try {
        const llmRes = await fetch(llmUrl, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            model: llmModel,
            messages,
            stream:      true,
            temperature: 0.7,
            max_tokens:  1024,
          }),
          signal: signal ?? AbortSignal.timeout(timeout),
        });

        if (!llmRes.ok) throw new Error(`LLM ${llmRes.status}: ${llmRes.statusText}`);

        const reader  = llmRes.body?.getReader();
        if (!reader) throw new Error('Geen response body ontvangen van LLM');

        const decoder = new TextDecoder();

        while (true) {
          const { done, value } = await reader.read();
          if (done) break;

          for (const line of decoder.decode(value, { stream: true }).split('\n')) {
            if (!line.startsWith('data: ')) continue;
            const raw = line.slice(6).trim();
            if (raw === '[DONE]') {
              controller.enqueue(encoder.encode('data: [DONE]\n\n'));
              controller.close();
              return;
            }
            try {
              const token: string = JSON.parse(raw)?.choices?.[0]?.delta?.content ?? '';
              if (token) emit({ token });
            } catch { /* ongeldige chunk → overslaan */ }
          }
        }

        controller.enqueue(encoder.encode('data: [DONE]\n\n'));
        controller.close();
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        console.error('[LLM] Streaming fout:', msg);
        emit({ error: msg });
        controller.close();
      }
    },
  });
}

/** Maakt een gesigneerd JWT-token aan (geldig 24 uur). */
export async function signToken(payload: UserPayload): Promise<string> {
  return new SignJWT({ ...payload })
    .setProtectedHeader({ alg: 'HS256' })
    .setIssuedAt()
    .setExpirationTime('24h')
    .sign(secret);
}

/** Verifieert een JWT-token. Gooit een fout bij ongeldig/verlopen token. */
export async function verifyToken(token: string): Promise<UserPayload> {

}

    const store = await cookies();

    if (!token) return null;
    return await verifyToken(token);
  } catch {
    return null;
  }
}

/**
 * lib/rateLimit.ts
 * In-memory rate limiting op basis van IP-adres.
 * Verwijdert verlopen entries automatisch elke 5 minuten.
 */

interface Entry {
  count:   number;
  resetAt: number;
}

const store = new Map<string, Entry>();

// Automatisch opschonen elke 5 minuten
setInterval(() => {
  const now = Date.now();
  for (const [key, entry] of store.entries()) {
    if (entry.resetAt < now) store.delete(key);
  }
}, 5 * 60_000);

/**
 * Controleert of een IP de limiet heeft overschreden.
 *
 * @param ip       - IP-adres van de aanvrager
 * @param limit    - Max aantal requests per window (standaard 30)
 * @param windowMs - Window in ms (standaard 60 seconden)
 */
export function checkRateLimit(
  ip: string,
  limit    = 30,
  windowMs = 60_000
): { allowed: boolean; remaining: number; resetAt: number } {
  const now = Date.now();
  const key = `rl:${ip}`;

  let entry = store.get(key);

  if (!entry || entry.resetAt < now) {
    entry = { count: 1, resetAt: now + windowMs };
    store.set(key, entry);
    return { allowed: true, remaining: limit - 1, resetAt: entry.resetAt };
  }

  entry.count++;
  return {
    allowed:   entry.count <= limit,
    remaining: Math.max(0, limit - entry.count),
    resetAt:   entry.resetAt,
  };
}

/** GET /api/vehicles */
export async function GET() {
  try {
    const pool   = await getPool();
    const result = await pool.request().query(`
      SELECT v.*, l.name AS locationName, l.city AS locationCity
      FROM   Vehicles v
      LEFT   JOIN RentalLocations l ON v.locationId = l.id
      ORDER  BY v.brand, v.model
    `);
    return NextResponse.json(result.recordset);
  } catch (err) {
    console.error('[Vehicles] GET mislukt:', err);
    return NextResponse.json({ error: 'Database-fout.' }, { status: 500 });
  }
}

/** POST /api/vehicles */
export async function POST(request: NextRequest) {
  const user = await getUser();
  if (user?.role !== 'admin')
    return NextResponse.json({ error: 'Niet geautoriseerd.' }, { status: 403 });

  try {
    const b = await request.json();
    if (!b.brand || !b.model || !b.type || b.dailyPrice == null)
      return NextResponse.json({ error: 'brand, model, type en dailyPrice zijn verplicht.' }, { status: 400 });

    const pool   = await getPool();
    const result = await pool.request()
      .input('brand',              sql.NVarChar(100),      b.brand.trim())
      .input('model',              sql.NVarChar(100),      b.model.trim())
      .input('type',               sql.NVarChar(50),       b.type)
      .input('seats',              sql.Int,                b.seats      ?? 5)
      .input('engineCC',           sql.Int,                b.engineCC   ?? null)
      .input('dailyPrice',         sql.Decimal(10, 2),     parseFloat(b.dailyPrice))
      .input('availabilityStatus', sql.NVarChar(20),       b.availabilityStatus ?? 'available')
      .input('locationId',         sql.Int,                b.locationId ?? null)
      .input('lastUpdated',        sql.DateTime,           new Date())
      .query(`
        INSERT INTO Vehicles
          (brand, model, type, seats, engineCC, dailyPrice, availabilityStatus, locationId, lastUpdated)
        OUTPUT INSERTED.*
        VALUES
          (@brand, @model, @type, @seats, @engineCC, @dailyPrice, @availabilityStatus, @locationId, @lastUpdated)
      `);

    return NextResponse.json(result.recordset[0], { status: 201 });
  } catch (err) {
    console.error('[Vehicles] POST mislukt:', err);
    return NextResponse.json({ error: 'Database-fout.' }, { status: 500 });
  }
}

/**
 * components/ChatWindow.tsx
 * Volledig interactief chatvenster met SSE-streaming.
 *
 * Features:
 * - Streaming antwoorden via Server-Sent Events
 * - Automatisch scrollen naar laatste bericht
 * - Berichtgeschiedenis per sessie (conversationId)
 * - Nieuw gesprek starten
 */
'use client';

import { useState, useRef, useEffect, useCallback } from 'react';
import { Send, RefreshCw, Loader2 }                 from 'lucide-react';
import { ChatMessage }                               from './ChatMessage';
import { Button }                                    from './ui/button';
import { cn }                                        from '@/lib/utils';

interface Message {
  id:        string;
  role:      'user' | 'assistant';
  content:   string;
  timestamp: string;
  streaming?: boolean;
}

export function ChatWindow() {
  const [messages,        setMessages]        = useState<Message[]>([]);
  const [input,           setInput]           = useState('');
  const [conversationId,  setConversationId]  = useState<string>('');
  const [isLoading,       setIsLoading]       = useState(false);
  const [error,           setError]           = useState('');

  const bottomRef  = useRef<HTMLDivElement>(null);
  const inputRef   = useRef<HTMLTextAreaElement>(null);
  const abortRef   = useRef<AbortController | null>(null);

  // Scroll naar het laatste bericht
  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);

  // Welkomstbericht bij eerste render
  useEffect(() => {
    setMessages([{
      id:        'welcome',
      role:      'assistant',
      content:   'Goeiedag! Ik ben de digitale assistent van PeterAllesweter Autoverhuur. Waarmee kan ik u vandaag helpen?\n\nU kunt mij vragen stellen over:\n• Beschikbaarheid van voertuigen\n• Huurprijzen\n• Onze 15 vestigingen in België\n• Extra opties en verzekeringen',
      timestamp: new Date().toISOString(),
    }]);
  }, []);

  const sendMessage = useCallback(async () => {
    const text = input.trim();
    if (!text || isLoading) return;

    // Annuleer eventuele lopende aanvraag
    abortRef.current?.abort();
    abortRef.current = new AbortController();

    const userMsg: Message = {
      id:        `u_${Date.now()}`,
      role:      'user',
      content:   text,
      timestamp: new Date().toISOString(),
    };

    setMessages(prev => [...prev, userMsg]);
    setInput('');
    setError('');
    setIsLoading(true);

    // Tijdelijk streaming-bericht aanmaken
    const botId = `b_${Date.now()}`;
    setMessages(prev => [...prev, {
      id:        botId,
      role:      'assistant',
      content:   '',
      timestamp: new Date().toISOString(),
      streaming: true,
    }]);

    try {
      const res = await fetch('/api/chat', {
        method:  'POST',
        headers: { 'Content-Type': 'application/json' },
        body:    JSON.stringify({ message: text, conversationId }),
        signal:  abortRef.current.signal,
      });

      const reader  = res.body?.getReader();
      if (!reader) throw new Error('Geen stream ontvangen');

      const decoder = new TextDecoder();
      let   buffer  = '';

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n');
        buffer = lines.pop() ?? '';

        for (const line of lines) {
          if (!line.startsWith('data: ')) continue;
          const raw = line.slice(6).trim();

          if (raw === '[DONE]') {
            setMessages(prev =>
              prev.map(m => m.id === botId ? { ...m, streaming: false } : m)
            );
            setIsLoading(false);
            return;
          }

          try {
            const parsed = JSON.parse(raw);

            if (parsed.error) {
              setError(parsed.error);
              setMessages(prev => prev.filter(m => m.id !== botId));
              setIsLoading(false);
              return;
            }

            if (parsed.conversationId && !conversationId) {
              setConversationId(parsed.conversationId);
            }

            if (parsed.token) {
              setMessages(prev =>
                prev.map(m =>
                  m.id === botId
                    ? { ...m, content: m.content + parsed.token }
                    : m
                )
              );
            }
          } catch { /* ongeldige JSON chunk */ }
        }
      }
    } catch (err) {
      if ((err as Error).name !== 'AbortError') {
        setError('Verbindingsfout. Controleer of LM Studio actief is.');
        setMessages(prev => prev.filter(m => m.id !== botId));
      }
    } finally {
      setIsLoading(false);
    }
  }, [input, isLoading, conversationId]);

  const handleKeyDown = (e: React.KeyboardEvent<HTMLTextAreaElement>) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      sendMessage();
    }
  };

  const startNewConversation = () => {
    abortRef.current?.abort();
    setConversationId('');
    setMessages([{
      id:        'welcome-new',
      role:      'assistant',
      content:   'Nieuw gesprek gestart. Hoe kan ik u helpen?',
      timestamp: new Date().toISOString(),
    }]);
    setError('');
    inputRef.current?.focus();
  };

  return (
    <div className="flex h-full flex-col rounded-2xl border border-gray-200 bg-white shadow-lg overflow-hidden">
      {/* Header */}
      <div className="flex items-center justify-between border-b border-gray-200 bg-brand-600 px-4 py-3">
        <div className="flex items-center gap-2">
          <div className="h-2.5 w-2.5 rounded-full bg-green-400 animate-pulse" />
          <span className="font-semibold text-white text-sm">PeterAllesweter Assistent</span>
        </div>
        <Button
          variant="ghost"
          size="icon"
          onClick={startNewConversation}
          title="Nieuw gesprek"
          className="text-white hover:bg-brand-700"
        >
          <RefreshCw size={16} />
        </Button>
      </div>

      {/* Berichten */}
      <div className="flex-1 overflow-y-auto chat-scroll py-2">
        {messages.map(msg => (
          <ChatMessage
            key={msg.id}
            role={msg.role}
            content={msg.content}
            timestamp={msg.timestamp}
            streaming={msg.streaming}
          />

function LocationsTab() {
  const [locations, setLocations] = useState<RentalLocation[]>([]);
  const [loading,   setLoading]   = useState(true);
  const [editing,   setEditing]   = useState<Partial<RentalLocation> | null>(null);
  const [isNew,     setIsNew]     = useState(false);
  const [saving,    setSaving]    = useState(false);
  const [error,     setError]     = useState('');

  const load = useCallback(async () => {
    setLoading(true);
    const res = await fetch('/api/locations');
    setLocations(await res.json());
    setLoading(false);
  }, []);

  useEffect(() => { load(); }, [load]);

  const startNew  = () => { setEditing(emptyLocation()); setIsNew(true); setError(''); };
  const startEdit = (l: RentalLocation) => { setEditing({ ...l }); setIsNew(false); setError(''); };
  const cancel    = () => { setEditing(null); setError(''); };

  const save = async () => {
    if (!editing) return;
    setSaving(true); setError('');
    try {
      const res = await fetch('/api/locations', {
        method:  isNew ? 'POST' : 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body:    JSON.stringify(editing),
      });
      if (!res.ok) { const d = await res.json(); setError(d.error); }
      else { setEditing(null); load(); }
    } catch { setError('Netwerkfout.'); }
    setSaving(false);
  };

  const remove = async (id: number) => {
    if (!confirm('Locatie deactiveren?')) return;

    load();
  };

## Architectuur

### Chat-aanvraagsflow

```
Browser
  │  POST /api/chat { message, conversationId }
  ▼
app/api/chat/route.ts
  ├─ 1. Rate-limit check (IP-gebaseerd, 30 req/min)
  ├─ 2. Input-validatie + sanitering (max 2000 tekens, geen HTML)
  ├─ 3. DB: gebruikersbericht opslaan
  ├─ 4. lib/llm.ts → buildSystemPrompt()
  │      └─ DB: getVehicles() + getLocations()  ← live data!
  │      └─ DB: logVehicleQuery()               ← analytics
  ├─ 5. DB: laatste 10 berichten laden (context)
  ├─ 6. LM Studio: POST met stream: true
  └─ 7. SSE tokens → browser (ReadableStream)
       └─ DB: volledig antwoord opslaan (na [DONE])
```

### Dynamische system-prompt

Elke chat-aanroep haalt live vlootdata op:
```
• Volkswagen Golf (hatchback) — €49/dag | ✅ BESCHIKBAAR | 5 zitpl. | 1598cc | Brussel
• Tesla Model 3 (electric) — €110/dag | ✅ BESCHIKBAAR | 5 zitpl. | Elektrisch | Zaventem
...
```

De LLM ziet altijd de meest actuele prijzen en beschikbaarheden.

### Beveiliging

- **JWT** in httpOnly cookie (tamper-proof)
- **Middleware** beschermt `/admin` routes server-side
- **Rate limiting** per IP (30 req/min voor chatendpoint)
- **Parameterized queries** in alle DB-aanroepen (geen SQL-injectie)
- **Input-sanitering**: max lengte + HTML-escape op gebruikersinput
- **Geen API-keys in code** — alles in `.env`

---

## Veelgestelde vragen

**LM Studio geeft geen antwoord?**
- Controleer dat LM Studio-server actief is
- Controleer `LLM_URL` in `.env` (IP + poort)
- Controleer dat een model geladen is in LM Studio

**SQL Server-verbinding mislukt?**
- Controleer `DB_SERVER`, `DB_USER`, `DB_PASSWORD` in `.env`
- Controleer dat SQL Server Browser-service actief is (voor named instances)
- Controleer firewallregels voor TCP/1433

**Admin-paneel niet toegankelijk?**
- Login met het admin-account (zie tabel hierboven)
- Controleer `JWT_SECRET` in `.env`

/**
 * Geeft true als het voertuig geen overlappende bevestigde/actieve huur heeft
 * én availabilityStatus = 'available'.
 */
export async function checkVehicleAvailability(
  vehicleId: number,
  startDateTime: Date,
  endDateTime: Date
): Promise<boolean> {
  const pool = await getPool();

  // Controleer eerst de status
  const statusResult = await pool.request()
    .input('vehicleId', sql.Int, vehicleId)
    .query(`SELECT availabilityStatus FROM Vehicles WHERE id = @vehicleId`);

  const status = statusResult.recordset[0]?.availabilityStatus;
  if (status !== 'available') return false;

  // Controleer overlappende verhuur
  const overlapResult = await pool.request()
    .input('vehicleId', sql.Int,      vehicleId)
    .input('start',     sql.DateTime, startDateTime)
    .input('end',       sql.DateTime, endDateTime)
    .query(`
      SELECT COUNT(*) AS cnt
      FROM   Rentals
      WHERE  vehicleId = @vehicleId
        AND  status    IN ('confirmed', 'active')
        AND  startDateTime < @end
        AND  endDateTime   > @start
    `);

  return overlapResult.recordset[0].cnt === 0;
}

function ConversationsTab() {
  const [conversations, setConversations] = useState<Conversation[]>([]);
  const [loading,       setLoading]       = useState(true);
  const [expanded,      setExpanded]      = useState<string | null>(null);
  const [messages,      setMessages]      = useState<Message[]>([]);
  const [loadingMsgs,   setLoadingMsgs]   = useState(false);

  useEffect(() => {
    fetch('/api/conversations')
      .then(r => r.json())
      .then(data => { setConversations(data); setLoading(false); });
  }, []);

  const toggleConversation = async (id: string) => {
    if (expanded === id) { setExpanded(null); return; }
    setExpanded(id);
    setLoadingMsgs(true);
    const res = await fetch(`/api/conversations?id=${encodeURIComponent(id)}`);
    setMessages(await res.json());
    setLoadingMsgs(false);
  };

  return (
    <div>
      <div className="mb-4">
        <h2 className="font-semibold text-gray-900">Chatgesprekken ({conversations.length})</h2>
      </div>

      {loading ? <p className="text-sm text-gray-500">Laden…</p> : (
        <div className="space-y-2">
          {conversations.map(c => (
            <div key={c.conversationId} className="rounded-xl border border-gray-200 bg-white overflow-hidden">
              <button
                className="w-full flex items-center justify-between px-4 py-3 text-left hover:bg-gray-50"
                onClick={() => toggleConversation(c.conversationId)}
              >
                <div className="flex items-center gap-3">
                  <MessageSquare size={16} className="text-brand-600 shrink-0" />
                  <div>
                    <p className="text-sm font-medium text-gray-900">{c.userId}</p>

# PeterAllesweter — Claude Code Project Guide

## Project Overview
Car rental chatbot website for PeterAllesweter, built with **Next.js 14 (App Router)**, **TypeScript**, **Tailwind CSS**, and a **Microsoft SQL Server** backend (`PeterAllesweterdb`). Includes an AI chatbot powered by a local LLM via LM Studio.

## Tech Stack
- **Framework**: Next.js 14 (App Router, `src/` dir not used)
- **Styling**: Tailwind CSS with custom Dockx color scheme
- **Database**: MS SQL Server via `mssql` package
- **LLM**: Local LM Studio API (OpenAI-compatible endpoint)
- **UI Components**: shadcn/ui (custom-themed)

## Environment Variables (`.env`)
| Variable | Purpose |
|---|---|
| `DB_SERVER` | SQL Server hostname\instance |
| `DB_DATABASE` | `PeterAllesweterdb` |
| `DB_USER` / `DB_PASSWORD` | SQL auth credentials |
| `LLM_URL` | LM Studio API endpoint |
| `LLM_MODEL` | Loaded model name in LM Studio |
| `LLM_TIMEOUT` | Request timeout (ms) |
| `CONTACT_EMAIL` | Email shown in UI |

## Database Schema (`PeterAllesweterdb`)
8 tables — always JOIN for current data:

```sql
EngineTypes       -- id, name, fuelType, co2Category
RentalLocations   -- id, name, address, ...
Vehicles          -- id, make, model, type, seats, licensePlate, year,
                  --   engineTypeId (FK), engineCC, powerKW, transmissionType,
                  --   mileage, availabilityStatus, locationId (FK), notes
VehiclePricing    -- id, vehicleId (FK), halfDayPrice, fullDayPrice,
                  --   weekendPrice, weekPrice, monthPrice,
                  --   validFrom, validUntil  (NULL = current price)
Rentals           -- id, vehicleId, conversationId, startDateTime, endDateTime,
                  --   status ('confirmed'|'active'|'completed'|'cancelled')
Conversations     -- id, sessionId, createdAt
Messages          -- id, conversationId, role, content, createdAt
VehicleQueries    -- id, conversationId, queryType, params, createdAt
```

**Key JOIN pattern** (always use `validUntil IS NULL` for current pricing):
```sql
LEFT JOIN VehiclePricing p ON v.id = p.vehicleId AND p.validUntil IS NULL
LEFT JOIN EngineTypes e ON v.engineTypeId = e.id
```

**Availability check**: `availabilityStatus = 'available'` AND no overlapping Rentals with `status IN ('confirmed','active') AND startDateTime < @end AND endDateTime > @start`

## Important Behaviours / Known Fixes
- **Chat history bug (fixed)**: In `app/api/chat/route.ts`, fetch conversation history BEFORE saving the new user message to DB. If reversed, the new message appears twice in the LLM context.
- **VehiclePricing updates**: When updating a vehicle's prices via PUT, close the old row (`validUntil = today`) and insert a new row — never update in place.
- **FK constraint on DELETE**: Delete `VehiclePricing` rows before deleting a `Vehicle`.
